# Terraform Labs

## Overview

Learn Infrastructure as Code with Terraform through hands-on labs covering resource management, state handling, modules, and production patterns.

## Prerequisites

- Terraform 1.0+ installed
- Cloud provider account (AWS, GCP, or Azure)
- Basic understanding of cloud concepts

### Required Setup

```bash
# Verify Terraform installation
terraform version

# Initialize a working directory
mkdir terraform-lab && cd terraform-lab
terraform init

# For GCP: Configure authentication
gcloud auth application-default login

# For AWS: Configure credentials
aws configure
```

## Lab 1: Basic Resource Management

**Objective**: Create, modify, and destroy cloud resources using Terraform
**Estimated Time**: 30 minutes
**Prerequisites**: Terraform installed, cloud provider configured

### Step-by-Step

#### 1. Create Basic Infrastructure

```bash
mkdir -p lab1 && cd lab1

# Create main.tf
cat > main.tf << 'EOF'
# Configure the Google Cloud provider
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = "your-project-id"
  region  = "us-central1"
}

# Create a Compute Instance
resource "google_compute_instance" "web_server" {
  name         = "terraform-web-server"
  machine_type = "e2-medium"
  zone         = "us-central1-a"

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
      size  = 20
    }
  }

  network_interface {
    network = "default"
    access_config {
      // Ephemeral public IP
    }
  }

  tags = ["http-server", "https-server"]

  metadata = {
    startup-script = <<-EOF
      #!/bin/bash
      apt-get update
      apt-get install -y nginx
      systemctl start nginx
    EOF
  }
}

# Output the instance IP
output "web_server_ip" {
  value = google_compute_instance.web_server.network_interface[0].access_config[0].nat_ip
}
EOF

# Initialize Terraform
terraform init

# Format and validate
terraform fmt
terraform validate
```

#### 2. Plan and Apply

```bash
# Create execution plan
terraform plan -out=tfplan

# Apply changes
terraform apply tfplan

# Or apply without plan file (with auto-approve)
terraform apply -auto-approve

# View current state
terraform show
terraform state list
```

#### 3. Modify and Update Resources

```bash
# Update the instance
cat > main.tf << 'EOF'
resource "google_compute_instance" "web_server" {
  name         = "terraform-web-server"
  machine_type = "e2-standard-2"  # Changed from e2-medium
  zone         = "us-central1-a"

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
      size  = 50  # Changed from 20
    }
  }

  network_interface {
    network = "default"
    access_config {
      // Ephemeral public IP
    }
  }

  tags = ["http-server", "https-server", "ssh"]  # Added "ssh" tag

  metadata = {
    startup-script = <<-EOF
      #!/bin/bash
      apt-get update
      apt-get install -y nginx
      systemctl start nginx
    EOF
    environment = "production"  # Added metadata
  }
}

output "web_server_ip" {
  value = google_compute_instance.web_server.network_interface[0].access_config[0].nat_ip
}
EOF

# Plan the changes
terraform plan -out=tfplan

# Apply the changes
terraform apply tfplan
```

#### 4. Destroy Resources

```bash
# Destroy all resources
terraform destroy

# Destroy with auto-approve
terraform destroy -auto-approve

# Verify state is empty
terraform show
```

### Validation

```bash
# Verify Terraform configuration
terraform validate

# Check state before applying
terraform show
terraform state list

# After apply, verify resources in cloud console
# GCP: gcloud compute instances list
# AWS: aws ec2 describe-instances
```

### Cleanup

```bash
cd .. && rm -rf lab1
```

## Lab 2: Variables, Outputs, and Functions

**Objective**: Use Terraform variables, outputs, and built-in functions for reusable configurations
**Estimated Time**: 35 minutes
**Prerequisites**: Terraform installed, cloud provider configured

### Step-by-Step

#### 1. Define Variables

```bash
mkdir -p lab2 && cd lab2

cat > variables.tf << 'EOF'
variable "project_id" {
  description = "GCP Project ID"
  type        = string
}

variable "region" {
  description = "GCP Region"
  type        = string
  default     = "us-central1"
}

variable "environment" {
  description = "Environment name"
  type        = string
  validation {
    condition     = contains(["dev", "staging", "prod"], var.environment)
    error_message = "Environment must be dev, staging, or prod."
  }
}

variable "machine_type" {
  description = "Machine type for instances"
  type        = map(string)
  default = {
    dev     = "e2-micro"
    staging = "e2-small"
    prod    = "e2-medium"
  }
}

variable "instance_tags" {
  description = "Network tags for instances"
  type        = list(string)
  default     = []
}
EOF

# Create terraform.tfvars
cat > terraform.tfvars << 'EOF'
project_id  = "your-project-id"
environment = "dev"
instance_tags = ["http-server"]
EOF

# Or use auto.tfvars (loads automatically)
cat > terraform.auto.tfvars << 'EOF'
project_id  = "your-project-id"
environment = "staging"
EOF
```

#### 2. Create Resources with Variables

```bash
cat > main.tf << 'EOF'
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = var.project_id
  region  = var.region
}

resource "google_compute_instance" "server" {
  count        = var.environment == "prod" ? 3 : 1
  name         = "${var.environment}-server-${count.index}"
  machine_type = var.machine_type[var.environment]
  zone         = "${var.region}-a"

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
      size  = 20
    }
  }

  network_interface {
    network = "default"
    access_config {
    }
  }

  tags = concat(var.instance_tags, ["${var.environment}"])

  labels = {
    environment = var.environment
    managed_by  = "terraform"
  }
}
EOF
```

#### 3. Define Outputs

```bash
cat > outputs.tf << 'EOF'
output "instance_ips" {
  description = "Public IP addresses of the servers"
  value       = google_compute_instance.server[*].network_interface[0].access_config[0].nat_ip
}

output "instance_names" {
  description = "Names of the instances"
  value       = google_compute_instance.server[*].name
}

output "machine_type_selected" {
  description = "Machine type being used"
  value       = var.machine_type[var.environment]
}

output "environment_info" {
  description = "Current environment details"
  value = {
    env      = var.environment
    region   = var.region
    instance_count = length(google_compute_instance.server)
  }
}
EOF
```

#### 4. Use Built-in Functions

```bash
cat > functions.tf << 'EOF'
# Local values for computed configuration
locals {
  common_tags = {
    project     = var.project_id
    environment = var.environment
    managed_by  = "terraform"
  }

  # Generate suffix from timestamp
  timestamp_suffix = formatdate("YYYYMMDD", timestamp())

  # Combine maps
  merged_labels = merge(
    local.common_tags,
    {
      "cost-center" = "engineering"
      "team"        = "devops"
    }
  )
}

# Update main.tf to use locals
EOF

cat > main.tf << 'EOF'
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = var.project_id
  region  = var.region
}

resource "google_compute_instance" "server" {
  count        = var.environment == "prod" ? 3 : 1
  name         = "${var.environment}-server-${count.index}-${local.timestamp_suffix}"
  machine_type = var.machine_type[var.environment]
  zone         = "${var.region}-a"

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
      size  = 20
    }
  }

  network_interface {
    network = "default"
    access_config {
    }
  }

  tags = concat(var.instance_tags, ["${var.environment}"])

  labels = local.merged_labels
}
EOF
```

### Validation

```bash
terraform init
terraform validate
terraform plan -out=tfplan
terraform apply tfplan

# View outputs
terraform output
terraform output instance_ips
terraform output -json
```

### Cleanup

```bash
terraform destroy -auto-approve
cd .. && rm -rf lab2
```

## Lab 3: State Management and Remote Backend

**Objective**: Configure remote state storage, state locking, and team collaboration
**Estimated Time**: 30 minutes
**Prerequisites**: Terraform installed, cloud provider configured

### Step-by-Step

#### 1. Configure Remote Backend (GCS)

```bash
mkdir -p lab3 && cd lab3

cat > main.tf << 'EOF'
terraform {
  # Remote backend with state locking
  backend "gcs" {
    bucket = "your-terraform-state-bucket"
    prefix = "prod/network"
  }

  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = "your-project-id"
  region  = "us-central1"
}

# Example resource
resource "google_compute_network" "vpc" {
  name                    = "terraform-vpc"
  auto_create_subnetworks = false
}
EOF

# Initialize with remote backend
terraform init
```

#### 2. Create State Bucket (if not exists)

```bash
# Create GCS bucket for state storage
gsutil mb -l us-central1 gs://your-terraform-state-bucket

# Enable versioning
gsutil versioning set on gs://your-terraform-state-bucket

# Enable encryption
gsutil encryption set-default --key [KMS_KEY] gs://your-terraform-state-bucket

# Bucket policy (prevent accidental deletion)
gsutil uniformbucketlevelaccess set on gs://your-terraform-state-bucket
```

#### 3. State Commands

```bash
# View current state
terraform show

# List resources in state
terraform state list

# List resources of a specific type
terraform state list google_compute_network

# Pull state to local file
terraform state pull > local-state.tfstate

# Push state to backend
terraform state push local-state.tfstate

# Rename resource in state
terraform state mv google_compute_network.vpc google_compute_network.main_vpc

# Remove resource from state (doesn't destroy cloud resource)
terraform state rm google_compute_network.vpc

# Show resource details
terraform state show google_compute_network.vpc
```

#### 4. Import Existing Resources

```bash
# Import existing infrastructure into state
# Find the resource address to use
terraform import google_compute_network.vpc your-project-id/terraform-vpc

# Or import with new resource name
terraform import 'google_compute_network.vpc' projects/your-project-id/global/networks/terraform-vpc

# Verify import
terraform state show google_compute_network.vpc
```

### Validation

```bash
# Verify state is stored remotely
terraform state list

# Lock state by running plan
terraform plan

# Verify in GCS console that state file exists
gsutil ls gs://your-terraform-state-bucket/prod/network/
```

### Cleanup

```bash
terraform destroy -auto-approve
# Note: Don't delete the state bucket as it may contain state for other resources
cd .. && rm -rf lab3
```

## Lab 4: Modules for Reusability

**Objective**: Create and use Terraform modules for reusable infrastructure patterns
**Estimated Time**: 40 minutes
**Prerequisites**: Terraform installed, cloud provider configured

### Step-by-Step

#### 1. Create a Module Structure

```bash
mkdir -p lab4/modules/gce-instance

cat > lab4/modules/gce-instance/main.tf << 'EOF'
variable "name" {
  description = "Instance name"
  type        = string
}

variable "machine_type" {
  description = "Machine type"
  type        = string
  default     = "e2-medium"
}

variable "zone" {
  description = "Zone for instance"
  type        = string
}

variable "project_id" {
  description = "GCP project ID"
  type        = string
}

variable "network" {
  description = "Network name"
  type        = string
  default     = "default"
}

variable "startup_script" {
  description = "Startup script"
  type        = string
  default     = ""
}

variable "labels" {
  description = "Labels to apply"
  type        = map(string)
  default     = {}
}

resource "google_compute_instance" "instance" {
  name         = var.name
  machine_type = var.machine_type
  zone         = var.zone
  project      = var.project_id

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
    }
  }

  network_interface {
    network    = var.network
    subnetwork = "default"
    access_config {
    }
  }

  metadata = {
    startup-script = var.startup_script
  }

  labels = var.labels
}

output "instance_ip" {
  value = google_compute_instance.instance.network_interface[0].access_config[0].nat_ip
}

output "instance_name" {
  value = google_compute_instance.instance.name
}
EOF

cat > lab4/modules/gce-instance/variables.tf << 'EOF'
variable "name" {
  type = string
}

variable "machine_type" {
  type    = string
  default = "e2-medium"
}

variable "zone" {
  type = string
}

variable "project_id" {
  type = string
}

variable "network" {
  type    = string
  default = "default"
}

variable "startup_script" {
  type    = string
  default = ""
}

variable "labels" {
  type    = map(string)
  default = {}
}
EOF
```

#### 2. Create Outputs

```bash
cat > lab4/modules/gce-instance/outputs.tf << 'EOF'
output "instance_ip" {
  description = "Public IP of the instance"
  value       = google_compute_instance.instance.network_interface[0].access_config[0].nat_ip
}

output "instance_name" {
  description = "Name of the instance"
  value       = google_compute_instance.instance.name
}

output "instance_self_link" {
  description = "Self link of the instance"
  value       = google_compute_instance.instance.self_link
}
EOF
```

#### 3. Use the Module

```bash
cat > lab4/main.tf << 'EOF'
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = "your-project-id"
  region  = "us-central1"
}

module "web_server" {
  source = "./modules/gce-instance"

  name        = "web-server"
  machine_type = "e2-medium"
  zone        = "us-central1-a"
  project_id  = "your-project-id"

  startup_script = <<-EOF
    #!/bin/bash
    apt-get update && apt-get install -y nginx
  EOF

  labels = {
    role = "web"
    env  = "production"
  }
}

module "api_server" {
  source = "./modules/gce-instance"

  name        = "api-server"
  machine_type = "e2-small"
  zone        = "us-central1-b"
  project_id  = "your-project-id"

  labels = {
    role = "api"
    env  = "production"
  }
}
EOF
```

#### 4. Module Versioning with Registry

```bash
cat > lab4/main-remote.tf << 'EOF'
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = "your-project-id"
  region  = "us-central1"
}

# Use module from Terraform Registry
module "vpc" {
  source  = "terraform-google-modules/network/google"
  version = "~> 7.0"

  project_id   = "your-project-id"
  network_name = "terraform-vpc"
  subnets = [
    {
      subnet_name   = "subnet-01"
      subnet_ip     = "10.10.10.0/24"
      subnet_region = "us-central1"
    },
    {
      subnet_name   = "subnet-02"
      subnet_ip     = "10.10.20.0/24"
      subnet_region = "us-east1"
    }
  ]
}
EOF
```

### Validation

```bash
cd lab4
terraform init
terraform validate
terraform plan -out=tfplan
terraform apply tfplan

# Verify instances
terraform output
gcloud compute instances list --filter="name~web-server OR name~api-server"
```

### Cleanup

```bash
terraform destroy -auto-approve
cd .. && rm -rf lab4
```

## Lab 5: Provisioners and Data Sources

**Objective**: Use provisioners for configuration management and data sources for referencing existing resources
**Estimated Time**: 35 minutes
**Prerequisites**: Terraform installed, cloud provider configured

### Step-by-Step

#### 1. Data Sources

```bash
mkdir -p lab5 && cd lab5

cat > data-sources.tf << 'EOF'
# Get information about the current project
data "google_project" "project" {}

# Get information about a specific network
data "google_compute_network" "default" {
  name = "default"
}

# Get latest Debian image
data "google_compute_image" "debian" {
  family  = "debian-11"
  project = "debian-cloud"
}

# Get available zones in a region
data "google_compute_zones" "available" {
  region = "us-central1"
}

output "project_id" {
  value = data.google_project.project.project_id
}

output "network_self_link" {
  value = data.google_compute_network.default.self_link
}
EOF
```

#### 2. Use Data Sources in Resources

```bash
cat > main.tf << 'EOF'
terraform {
  required_providers {
    google = {
      source  = "hashicorp/google"
      version = "~> 4.0"
    }
  }
}

provider "google" {
  project = "your-project-id"
  region  = "us-central1"
}

data "google_project" "project" {}

data "google_compute_image" "debian" {
  family  = "debian-11"
  project = "debian-cloud"
}

resource "google_compute_instance" "server" {
  name         = "data-source-server"
  machine_type = "e2-medium"
  zone         = "us-central1-a"

  boot_disk {
    initialize_params {
      image = data.google_compute_image.debian.self_link
    }
  }

  network_interface {
    network    = "default"
    access_config {
    }
  }

  labels = {
    project_id = data.google_project.project.project_id
  }
}
EOF
```

#### 3. Provisioners

```bash
cat > provisioners.tf << 'EOF'
# Example with local-exec provisioner
resource "null_resource" "example" {
  triggers = {
    always = timestamp()
  }

  provisioner "local-exec" {
    command = "echo 'Server ${google_compute_instance.server.name} created at ${timestamp()}' >> created_servers.txt"
  }
}

# File provisioner example
resource "google_compute_instance" "with_config" {
  name         = "configured-server"
  machine_type = "e2-medium"
  zone         = "us-central1-a"

  boot_disk {
    initialize_params {
      image = "debian-cloud/debian-11"
    }
  }

  network_interface {
    network    = "default"
    access_config {
    }
  }

  # Example: fetch configuration from remote
  provisioner "file" {
    content     = templatefile("${path.module}/templates/app.conf.tpl", { env = "production" })
    destination = "/tmp/app.conf"
  }

  # Run configuration script
  provisioner "remote-exec" {
    inline = [
      "chmod +x /tmp/setup.sh",
      "/tmp/setup.sh",
    ]
  }

  connection {
    type        = "ssh"
    user        = "terraform"
    private_key = file("~/.ssh/id_rsa")
    host        = self.network_interface[0].access_config[0].nat_ip
  }
}

# Create template file
mkdir -p lab5/templates
cat > lab5/templates/app.conf.tpl << 'TPL'
environment = "${env}"
log_level = "info"
TPL
```

### Validation

```bash
terraform init
terraform validate
terraform plan
terraform apply -auto-approve

# Verify data sources
terraform output

# Check provisioner ran
cat created_servers.txt
```

### Cleanup

```bash
terraform destroy -auto-approve
rm -rf lab5/templates created_servers.txt
cd .. && rm -rf lab5
```
